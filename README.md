# My Basic Git Commands

- git init
- git add . 
- git add <file_name>
- git switch <branch_name>
- git branch
- git branch <branch_name>
- git merge <branch_name_to_merge_from>
- git commit -m "<message>"
- git push 
- git push -u origin <branch_name>
- git remote
- git remote add origin <url>
- git remote remove origin
- git log
